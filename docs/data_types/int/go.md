# Go

## Int

```go
package main

import "fmt"

func main() {
	var a int = 10
	var b int8 = 10
	var c int16 = 10
	var d int32 = 10
	var e int64 = 10
	var f uint = 10
	var g uint8 = 10
	var h uint16 = 10
	var i uint32 = 10
	var j uint64 = 10

	fmt.Printf("a = %d\n", a)
	fmt.Printf("b = %d\n", b)
	fmt.Printf("c = %d\n", c)
	fmt.Printf("d = %d\n", d)
	fmt.Printf("e = %d\n", e)
	fmt.Printf("f = %d\n", f)
	fmt.Printf("g = %d\n", g)
	fmt.Printf("h = %d\n", h)
	fmt.Printf("i = %d\n", i)
	fmt.Printf("j = %d\n", j)
}
```
